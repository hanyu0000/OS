```bash
{
    # 使用 nasm 编译 mbr.S 文件
    nasm -f bin -o build/mbr chapter2/mbr.S
    # 使用 dd 命令将 mbr 文件写入 HD60.img 文件
    dd if=build/mbr of=HD60.img bs=512 count=1 conv=notrunc
}
```
